-- 3.1. Самая популярная категория товаров
top_product_categories AS (
    SELECT 
        customer_id, report_period, product_type
    FROM (
        SELECT 
            customer_id, report_period, product_type,
            ROW_NUMBER() OVER(PARTITION BY customer_id, report_period ORDER BY COUNT(product_id) DESC) as rnk
        FROM dwh_delta
        GROUP BY customer_id, report_period, product_type
    ) t WHERE rnk = 1
),

-- 3.2. Самый популярный мастер
top_craftsmans AS (
    SELECT 
        customer_id, report_period, craftsman_id
    FROM (
        SELECT 
            customer_id, report_period, craftsman_id,
            ROW_NUMBER() OVER(PARTITION BY customer_id, report_period ORDER BY COUNT(order_id) DESC) as rnk
        FROM dwh_delta
        GROUP BY customer_id, report_period, craftsman_id
    ) t WHERE rnk = 1
),

-- 3.3. Итоговый расчет агрегатов с джойном топов
calculated_data AS (
    SELECT
        T1.customer_id, T1.customer_name, T1.customer_address, T1.customer_birthday, T1.customer_email, T1.report_period,
        SUM(T1.product_price) AS customer_money,
        SUM(T1.product_price) * 0.1 AS platform_money,
        COUNT(order_id) AS count_order,
        AVG(T1.product_price) AS avg_price_order,
        PERCENTILE_CONT(0.5) WITHIN GROUP(ORDER BY diff_order_date) AS median_time_order_completed,
        tpc.product_type AS top_product_category,
        tc.craftsman_id AS top_craftsman_id,
        SUM(CASE WHEN T1.order_status = 'created' THEN 1 ELSE 0 END) AS count_order_created,
        SUM(CASE WHEN T1.order_status = 'in progress' THEN 1 ELSE 0 END) AS count_order_in_progress,
        SUM(CASE WHEN T1.order_status = 'delivery' THEN 1 ELSE 0 END) AS count_order_delivery,
        SUM(CASE WHEN T1.order_status = 'done' THEN 1 ELSE 0 END) AS count_order_done,
        SUM(CASE WHEN T1.order_status != 'done' THEN 1 ELSE 0 END) AS count_order_not_done
    FROM dwh_delta T1
    LEFT JOIN top_product_categories tpc ON T1.customer_id = tpc.customer_id AND T1.report_period = tpc.report_period
    LEFT JOIN top_craftsmans tc ON T1.customer_id = tc.customer_id AND T1.report_period = tc.report_period
    GROUP BY T1.customer_id, T1.customer_name, T1.customer_address, T1.customer_birthday, T1.customer_email, T1.report_period, tpc.product_type, tc.craftsman_id
)
